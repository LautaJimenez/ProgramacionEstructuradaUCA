void CargarEquiposDinamicos(char **Equipo){

    FILE *ArchivoEquipos = fopen("equipos.txt","r");
    
    char Caracter;
    
    *Equipo = malloc(sizeof(char));

    int i = 0;
    Caracter = fgetc(ArchivoEquipos);

    while(Caracter != EOF){

        *(*Equipo + i) = Caracter;
        i++;
        *Equipo = realloc(*Equipo, sizeof(char) * (i+1));

        Caracter = fgetc(ArchivoEquipos);
    }

    *(*Equipo + i) = '\0';

    fclose(ArchivoEquipos);
}

t_equipo CargarEquipo(){

    t_equipo EstructuraEquipo;

    char *Equipo;

    CargarEquiposDinamicos(&Equipo);

    EstructuraEquipo.Equipo = Equipo;

    return EstructuraEquipo;

}

    t_equipo EquipoNuevo;
